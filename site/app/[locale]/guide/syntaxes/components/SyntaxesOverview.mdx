export default {
    syntaxes: {
        foo: {
            matcher: /^foo:./,
            unit: 'rem',
            declare(value, unit) {
                return {
                    width: value + unit
                }
            }
        }
    }
}

```html
<div class="**foo:32**"></div>
```
