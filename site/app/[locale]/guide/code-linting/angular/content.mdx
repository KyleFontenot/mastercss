<StepSection className="mt:12x">
    <Step $row>
        <StepL>
            ### <StepNum />Installation
            Install the plugin and parser into your project via package managers.
        </StepL>
        <StepR>
            ```install
            -D @master/eslint-config-css@rc @angular-eslint/template-parser @typescript-eslint/parser eslint
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Set up ESLint config
            Import and add the Master CSS ESLint Configuration `html css` to [`eslint.config.mjs`](https://eslint.org/docs/latest/use/getting-started).

            - Use `@angular-eslint/template-parser` to parse `.html` files.
            - Use `@typescript-eslint/parser` to parse `.tsx` files.
        </StepL>
        <StepR>
            ```js name=eslint.config.mjs
            import htmlParser from "@angular-eslint/template-parser"
            import tsParser from '@typescript-eslint/parser'
            import css from '@master/eslint-config-css'

            export default [
                {
                    files: ['**/*.html'],
                    languageOptions: { parser: htmlParser }
                },
                {
                    files: ['**/*.ts'],
                    languageOptions: { parser: tsParser }
                },
                css
            ]
            ```
        </StepR>
    </Step>
    <Step>
        ### <StepNum />Start linting your code
        ```bash
        npx eslint
        ```
    </Step>
</StepSection>
