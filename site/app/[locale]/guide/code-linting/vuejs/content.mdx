            ### <StepNum />Set up ESLint config
            Import and add the Master CSS ESLint Configuration `html css` to [`eslint.config.mjs`](https://eslint.org/docs/latest/use/getting-started).

            ```js name=eslint.config.mjs
            import { includeIgnoreFile } from '@eslint/compat'
            import { fileURLToPath } from 'node:url'
            import vue from 'eslint-plugin-vue'
            import vueTS from '@vue/eslint-config-typescript'
            +import css from '@master/eslint-config-css'

            const gitignorePath = fileURLToPath(new URL("./.gitignore", import.meta.url))

            export default [
                includeIgnoreFile(gitignorePath),
                ...vue.configs['flat/essential'],
                ...vueTS(),
            +    css
            ]
            ```
        </StepL>
    </Step>
    <Step>
        ### <StepNum />Start linting your code
        ```bash
        npx eslint
        ```
    </Step>
