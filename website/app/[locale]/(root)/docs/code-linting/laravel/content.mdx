<StepSection className="mt:12x">
    <Step $row>
        <StepL>
            ### <StepNum />Installation
            Install the plugin and parser into your project via package managers.
        </StepL>
        <StepR>
            ```install
            -D @master/eslint-config-css@rc @angular-eslint/template-parser eslint
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Set up your config
            Create [`.eslintrc.json`](https://eslint.org/docs/latest/use/getting-started) if you don't have one.
            - Add `@master/css` to the `extends` section.
            - Add `@angular-eslint/template-parser` to the `overrides` section.
        </StepL>
        <StepR>
            ```json name=.eslintrc.json
            {
                "extends": ["@master/css"],
                "overrides": [
                    {
                        "files": ["*.blade.php"],
                        "parser": "@angular-eslint/template-parser"
                    }
                ]
            }
            ```
        </StepR>
    </Step>
    <Step>
        ### <StepNum />Start linting your code
        ```bash
        npx eslint *.blade.php
        ```
    </Step>
</StepSection>
